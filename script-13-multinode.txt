pipeline {
	
		agent {
			label {
					label "built-in"
					customWorkspace "/mnt/master"
			}
		}
		
		stages {
			
			stage ("on-master"){
				steps {
						sh "yum install tree -y"
				}
			}
			
			stage ("on-slave-1"){
			agent {
			
			label {
			
				label "slave-1"
				customWorkspace "/mnt/pipeline"
			}
			}
			steps {
				sh "sudo yum install httpd -y"
				sh "sudo yum install tree -y"
			}
			}
			
			stage ("on-slave-2"){
			agent {
			label {
				label "slave-2"
				customWorkspace "/mnt/pipeline"
			}
			}
			steps {
				sh "sudo yum install httpd -y"
				sh "sudo yum install tree -y"
			}
			}
		
		}

}
